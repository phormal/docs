import { Phormal, useLength, useEmail, useRequired, useAutoCapitalize } from '@phormal/core';

const formFields = {
  firstName: {
    label: 'First name',
    hooks: [useRequired(), useLength(3, 155)]
  },
  lastName: {
    label: 'Last name',
    disabledIf: { firstName: 'empty' },
    placeholder: 'Enter your last name',
    hooks: [useAutoCapitalize()]
  },
  email: {
    label: 'Email',
    type: 'email',
    hooks: [useEmail()],
    value: 'john@doe.com'
  },
  password: {
    label: 'Password',
    type: 'password'
  },
  age: {
    label: 'Age',
    type: 'number',
    value: 18,
    disabled: true
  },
  year: {
    label: 'Birthyear',
    type: 'number',
    row: 'birthdate'
  },
  month: {
    label: 'Month',
    type: 'number',
    row: 'birthdate'
  },
  day: {
    label: 'Day',
    type: 'number',
    row: 'birthdate'
  }
}

const formConfig = { el: '#phormal', theme: /** name of theme */ }

const form = new Phormal(formFields, formConfig)
